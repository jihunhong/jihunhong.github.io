class Solution {
    public int solution(int X, int[] A) {
        // write your code in Java SE 8
        for (int i=0 ; i<A.length; i++){
            for(int j=0; j<A.length; j++){
                if(A[i]==A[j] && i!=j && A[i]!=0)
                {
                    A[j] = 0;
                }
            }
        }
        
        int sum = (X*(X+1))/2;
        int flag = 0;
        
        for (int i=0; i<A.length; i++){
 
            sum -= A[i];
            if (sum == 0){
                flag = i;
                i = A.length;
            }
        }
        if(sum >0)
        {
            return -1;
        }
        
        return flag;
    }
}
