       b.salary
FROM   employees a,
       salaries b
WHERE  a.emp_no = b.emp_no
       AND b.to_date = '9999-01-01'
       AND b.salary < (SELECT Avg(salary)
                       FROM   salaries
                       WHERE  to_date = '9999-01-01');

SELECT Min(avg_salary)
FROM   (SELECT Round(Avg(a.salary)) AS avg_salary
        FROM   salaries a,
               titles b
        WHERE  a.emp_no = b.emp_no
               AND a.to_date = '9999-01-01'
               AND b.to_date = '9999-01-01'
