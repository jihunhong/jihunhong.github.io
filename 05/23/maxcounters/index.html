class Solution {
    public int[] solution(int N, int[] A) {
        // write your code in Java SE 8
        int B[] = new int [N];
        int counter = 0;
        
        for(int i=0; i<B.length; i++){
            B[i] = 0;
        }
        
        for(int i=0 ; i<A.length ; i++){
            
            if(A[i]==N+1){
                for(int j=0; j<B.length; j++){
                    B[j] = counter;
                }
            } else {
                B[A[i]-1] ++;
                counter = Math.max(B[A[i]-1],counter);
            }
        }
        return B;
    }
}
